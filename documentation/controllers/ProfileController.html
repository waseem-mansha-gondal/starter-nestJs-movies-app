import {
  Controller,
  Param,
  Post,
  UploadedFile,
  UseGuards,
  UseInterceptors,
  Request,
  Get,
  Patch,
} from '@nestjs/common';
import { FileInterceptor } from '@nestjs/platform-express';
import { JwtAuthGuard } from '../auth/guards/jwt-guards';
import {
  editFileName,
  imageFileFilter,
  ProfileService,
} from './profile.service';
import { diskStorage } from 'multer';
import { User } from '../users/user.entity';

@Controller('auth/user/profile')
export class ProfileController {
  constructor(private profileServices: ProfileService) {}
  @UseGuards(JwtAuthGuard)
  @Post()
  async createProfile(@Request() req) {
    return await this.profileServices.creatProfile(req);
  }

  @UseGuards(JwtAuthGuard)
  @Post('profilePic/:id')
  @UseInterceptors(
    FileInterceptor('image', {
      storage: diskStorage({
        destination: './uploads',
        filename: editFileName,
      }),
      fileFilter: imageFileFilter,
    }),
  )
  async uploadedFile(@Param() param, @UploadedFile() file) {
    // const user = await User.findOne({where:{id:req.user.id}})

    console.log(param.id);
    const profile = await this.profileServices.getProfile(param.id);
    profile.profilePicture = file.filename;
    await profile.save();
    const response = {
      originalname: file.originalname,
      filename: file.filename,
    };

    return response;
  }

  @Get()
  async getProfile(@Request() req) {
    return await this.profileServices.getProfile(req.user.id);
  }
  @Patch()
  async updateProfile(@Request() req) {
    return await this.profileServices.updateProfile(req.body, req.user.id);
  }
}
