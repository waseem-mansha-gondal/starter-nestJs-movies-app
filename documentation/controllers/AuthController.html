import {
  Controller,
  Body,
  Post,
  UseGuards,
  Request,
  Get,
  Req,
  HttpStatus,
  Param,
} from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { AuthService } from './auth.service';
import { UserDto } from '../users/dto/user.dto';
import { JwtAuthGuard } from './guards/jwt-guards';
import { hasRoles } from './decorators/roles.decorators';
import { RolesGuard } from './guards/roles-guards';
import { JwtService } from '@nestjs/jwt';
import { Role } from '../roles.enum';

@Controller('auth')
export class AuthController {
  constructor(
    private authService: AuthService,
    private jwtService: JwtService, // public let array=['admin','user']
  ) {}

  @UseGuards(AuthGuard('local'))
  @Post('login')
  async login(@Request() req) {
    return await this.authService.login(req.user);
  }

  @Post('signup')
  async signUp(@Body() user: UserDto) {
    return await this.authService.create(user);
  }
  @Get('signup/verification/:token')
  async verification(@Param() params) {
    return await this.authService.verification(params.token);
  }
  // let array=['user','admin'];
  @UseGuards(JwtAuthGuard, RolesGuard)
  @hasRoles(Role.User)
  @Get('users')
  async getUsers(@Request() req) {
    console.log(req.user);

    return await this.authService.getUsers();
  }
  @Get('google')
  @UseGuards(AuthGuard('google'))
  async googleAuth(@Req() req) {}

  @Get('google/redirect')
  @UseGuards(AuthGuard('google'))
  googleAuthRedirect(@Req() req) {
    return this.authService.googleLogin(req);
  }
  @Get('/facebook')
  @UseGuards(AuthGuard('facebook'))
  async facebookLogin(): Promise<any> {
    return HttpStatus.OK;
  }
  @Get('/facebook/redirect')
  @UseGuards(AuthGuard('facebook'))
  async facebookLoginRedirect(@Req() req: any): Promise<any> {
    return this.authService.faceBookLogin(req);
  }
}
